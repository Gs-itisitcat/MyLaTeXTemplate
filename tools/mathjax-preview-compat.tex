% Simple MathJax-safe fallback for xmat-style commands (optional keys ignored).
\newcommand{\xmat}[4][]{%
  \begin{matrix}
    #2_{1,1} & \cdots & #2_{1,#4} \\
    \vdots & \ddots & \vdots \\
    #2_{#3,1} & \cdots & #2_{#3,#4}
  \end{matrix}%
}
\newcommand{\pxmat}[4][]{\left( \xmat[#1]{#2}{#3}{#4} \right)}
